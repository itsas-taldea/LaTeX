\section{Área de investigación}
	\subsection{Robótica}
		\begin{frame}
			\frametitle{Robótica en entornos no estructurados}
			Entorno no estructurado: representación inviable
			\begin{itemize}
				\item por falta de información,
				\item porque el entorno es cambiante
				\item descripción no manejable por complicada.
			\end{itemize}
			Robótica Evolutiva adecuada (diseño automático)
		\end{frame}
		\begin{frame}
			\frametitle{\textit{Espectro de control robótico}}
			\textit{Espectro de control robótico}~\citep{arkin:98a}: sistemas deliberativos
				\begin{itemize}
					\item simbólicos,
					\item basados en Inteligencia Artificial ``clásica'' y
					\item adecuados para entornos altamente estructurados,
				\end{itemize}
			a sistemas reactivos
				\begin{itemize}
					\item subsimbólicos,
					\item basados en sistemas conexionistas y
					\item adecuados para entornos no estructurados.
				\end{itemize}
		\end{frame}
		\begin{frame}
			\frametitle{Conceptos para la Robótica}
				\begin{description}[Simulaciones mínimas:]
					\item [Corporeidad:] el controlador depende de la morfología del robot~\citep{bro:Repr}
					\item [Ubicación:] un robot debe estar acoplado a su entorno y tomarlo como el modelo del mundo~\citep{bro:Repr}
					\item [Simulaciones mínimas:] se pueden construir simulaciones sencillas que produzcan controladores válidos para robots físicos~\citep{jak}
				\end{description}
		\end{frame}
		\begin{frame}
			\frametitle{Diseño automático}
			Diseño \textit{a mano}: el programador elige
				\begin{itemize}
					\item la manera de procesar los datos
					\item determina qué información es relevante
				\end{itemize}

			Diseño automático: el sistema facilita
				\begin{itemize}
					\item la evolución el conjunto de hardware y controlador
					\item la aplicación de los conceptos de corporeidad y ubicación
					\item la decisión de la estructura perceptiva
				\end{itemize}
		\end{frame}
	\subsection{Entorno}
		\begin{frame}
			\frametitle{Inspiración Biológica}
			Referente en el diseño de robots autónomos
				\begin{itemize}
					\item funcional
					\item estructural
					\item con diferentes niveles de inspiración
				\end{itemize}

			Robótica Biomimética: copia estructural y funcional
		\end{frame}
		\begin{frame}
			\frametitle{Entorno multidisciplinar}
			Referente teórico para objetivo técnico, no descriptivo:
				\begin{description}[Filosofía de la Ciencia:]
					\item [Biología:] evolución y morfología
					\item [Neurociencias:] estructura y su relación con la percepción
					\item [Psicología:] percepción básica y comportamiento
					\item [Filosofía de la Ciencia:] visión de conjunto y abstracción de los mecanismos
				\end{description}
			Prioridad del objetivo práctico frente al conocimiento
		\end{frame}
		\begin{frame}
			\frametitle{Inteligencia, Emergencia, Autoorganización y Vida Artificial}
				\begin{block}{La \textit{Inteligencia} a través de la Vida Artificial}
					\begin{itemize}
						\item Comportamiento funcional complejo
						\item resultado de la dinámica global emergente
						\item interacción de multitud de elementos
						\item mediante reglas locales simples
						\item no reducible al comportamiento de las partes
					\end{itemize}
				\end{block}
		\end{frame}
	\subsection{Símbolos}
		\begin{frame}
			\frametitle{Símbolos o Base Física}
			\begin{block}{La \textit{Hipótesis del Sistema de Símbolos}}
				\begin{itemize}
					\item Los sensores suministran símbolos.
					\item Por ello el sistema de razonamiento puede separarse del sensomotor.
				\end{itemize}
			\end{block}
			\begin{block}{La \textit{Hipótesis de la Base Física}}
				\begin{itemize}
					\item Los sistemas deben usar representaciones basadas en el mundo físico: \textit{``Es mejor usar el mundo como su propio modelo"}~\citep{bro:Repr} % It turns out to be better to use the world as its own model
					\item La Inteligencia es inseparable del cuerpo y del entorno.
				\end{itemize}
			\end{block}
		\end{frame}
	\subsection{Conexionismo}
		\begin{frame}
			\frametitle{Conexionismo, PDP}
			Procesamiento Distribuido Paralelo: procesamiento subsimbólico de la información por un conjunto de elementos simples con reglas locales, de los que emerge una dinámica global.

			Inspirados en:
				\begin{description}[Neurociencias:]
					\item [Física:] Enfriamiento Estadístico (\textit{Simulated Annealing})
					\item [Topología:] SOM o Redes de Kohonen
					\item [Probabilidad:] Redes Bayesianas
					\item [Neurociencias:] Redes Neuronales
				\end{description}
		\end{frame}

\section{Robótica Evolutiva}
	\subsection{Objetivos y situación actual}
		\begin{frame}
			\frametitle{Objetivos de la Robótica Evolutiva}
			La Robótica Evolutiva
			\begin{itemize}
				\item estudia mecanismos cognitivos creando modelos
				\item crea sistemas con comportamientos funcionalmente más complejos
				\item encuentra mejores soluciones para entornos no estructurados mediante
				\begin{itemize}
					\item sistemas subsimbólicos
					\item diseño automático
				\end{itemize}
			\end{itemize}
		\end{frame}
	\subsection{Líneas de trabajo}
		\begin{frame}
			\frametitle{Líneas de trabajo en la Robótica Evolutiva}
			\begin{itemize}
				\item Estudio de los procesos cognitivos~\citep*{har:dip:woo:quin:tuc}.
				\begin{itemize}
					\item Validación de teorías neurocientíficas y biológicas, como la \textit{Fonotaxia del grillo}~\citep*{Webb,lun:web:hal}.
					\item Estudio procesos cognitivos con objetivos técnicos~\citep{suzuki08, suzukiT}.
				\end{itemize}
				\item Estudio de los modelos como sistemas dinámicos~\citep{Beer92b}.
			\end{itemize}
		\end{frame}
	\subsection{Simulaciones e implementaciones}
		\begin{frame}
			\frametitle{Simulaciones e implementaciones en robots físicos}
			Se han definido dos líneas en cuanto a las simulaciones:
			\begin{itemize}
				\item es inútil trabajar sólo en simulación porque no se asegura su funcionamiento en la realidad~\citep{flo:mon:96}.
				\item es posible definir una \textit{simulación mínima} que permite o facilita saltar de simulación a realidad~\citep{jak}.
			\end{itemize}
		\end{frame}
	\subsection{Límites y problemas}
		\begin{frame}
			\frametitle{Límites y problemas de la Robótica Evolutiva}
				Dos problemas principales:
				\begin{description}[Morfogénesis:]
					\item [Morfogénesis:] desarrollo de una forma a partir de la información genética
					\item [\textit{Escalabilidad}:] aplicación del mecanismo a problemas cuantitativamente más complejos
				\end{description}
		\end{frame}

	\subsection{Problema planteado}
		\begin{frame}
			\frametitle{Problema implementado en la aplicación {\scshape Topos}}
			Simulación de un robot que:
				\begin{itemize}
					\item navega usando puntos de referencia basados en sonido real
					\item distingue dos señales compuestas que varían en el tiempo (sonido real)
					\item recibe una señal que varía independientemente de su movimiento
					\item elige una de las dos señales acercándose a ella en un esquema \textit{caja de Skinner}
					\item percibe y actúa a través de una Red Neuronal de Pulsos y sensores evolucionados
					\item se diseña con un enfoque fuertemente bioinspirado
				\end{itemize}
		\end{frame}

	\subsection{Algoritmo Genético}
		\begin{frame}
			\frametitle{Algoritmo Genético}
			La representación del individuo en el genotipo es directa.

			Los parámetros del robot se codifican en números reales y enteros, y booleanos codificados como enteros.

			La evolución es elitista:
			\begin{itemize}
				\item el 25\% mejor: pasa a la siguiente generación.
				\item el 25\% peor: se desecha.
				\item el 75\% mejor (élite incluida): cruce y creación de la siguiente generación.
			\end{itemize}
			Mutación: 10\% de las copias de datos (gaussiana de media 1).
		\end{frame}
	\subsection{Fitness}
		\begin{frame}
			\frametitle{El problema y la función de adecuación o \textit{fitness function}}
			El problema se considera complejo pero la función de adecuación es sencilla:
			\begin{itemize}
				\item resta un valor proporcional al cuadrado de la distancia mínima al faro correcto
				\item suma un valor proporcional al cuadrado de la distancia mínima al faro incorrecto
				\item se suma (resta) una bonificación si se ha acercado a una distancia del faro.
				\item el valor final es la media de cinco pruebas, sustituyendo los negativos por cero.
			\end{itemize}
		\end{frame}

\section{Resultados}
	\subsection{Evaluación absoluta}
		\begin{frame}
			\frametitle{Resultados de los experimentos realizados con {\scshape Topos}}
				Evaluación: acertar, fallar o no alcanzar una fuente de sonido.

				Valores que indican la capacidad de hacer la tarea en porcentaje de efectividad.

				\begin{description}[eficacia absoluta:]
					\item [eficacia absoluta:] efa = aciertos/pruebas

\section{Conclusiones}
	\subsection{Conclusiones globales}
		\begin{frame}
			\frametitle{Resultados: viabilidad de un desarrollo técnico en la Robótica Evolutiva}
			Se ha construido una simulación en la que se obtiene mediante estrategias evolutivas un robot que:
				\begin{itemize}
					\item navega usando puntos de referencia complejos
					\item distingue y elige una de las dos señales compuestas que varían en el tiempo (sonido real)
					\item recibe una señal que varía independientemente de su movimiento
% 					\item  señales acercándose a ella en un esquema \textit{caja de Skinner}
					\item percibe y actúa a través de una Red Neuronal de Pulsos y sensores evolucionados
					\item se diseña con un enfoque fuertemente bioinspirado
				\end{itemize}
		\end{frame}
		\begin{frame}
			\frametitle{Conclusiones}
			El robot tiene un comportamiento de navegación al diferenciar sonidos variables en el tiempo que usa como puntos de referencia.

			El resultado valida el método como adecuado para obtener robots que desarrollen tareas \textit{complejas} en entornos no estructurados.

			El controlador del robot es mucho más sencillo que lo que haría falta con técnicas de IA clásica.
		\end{frame}
		\begin{frame}
			\frametitle{Dificultades comprobadas en la construcción del sistema}
			La aplicación tiene un diseño \textit{tradicional} en la Robótica Evolutiva.

			Se ha comprobado que este tipo de diseño dificulta las siguientes tareas:
				\begin{itemize}
					\item es difícil conseguir una descripción adecuada de un robot con esta complejidad para su uso evolutivo.
					\item el robot resultante no es suficientemente flexible para adaptarse a nuevas tareas o planteamientos.
				\end{itemize}
			Estas dificultades forman la base del trabajo futuro.
		\end{frame}


	\subsection{Trabajo futuro}
		\begin{frame}
			\frametitle{Trabajo futuro}
				\begin{description}[Morfogénesis:]
					\item [Morfogénesis:]
						\begin{itemize}
							\item expresión no lineal de la información genética
							\item sensores (exteroceptores y propioceptores) y motores complejos
							\item topología de la red neuronal más variada
							\item neuronas más potentes y mecanismos de aprendizaje
						\end{itemize}
					\item [\textit{Escalabilidad}:] los entornos complejos y la morfogénesis permitirán tareas complejas y estructuras inalcanzables con diseño a mano.
				\end{description}
		\end{frame}
